//จะเห็นแค่ SName Scode Mname Cname Ccode และแสดงแค่คนเดียว
SELECT student.SName,student.SCode,major.MName ,class.CName,class.CCode
FROM student 
JOIN major 
ON student.MId = major.MId

JOIN classregister
ON student.SId = classregister.SId

JOIN class
ON classregister.CId = class.CId
WHERE student.SId = 1 //แสดงแค่คนเดียว
ORDER BY student.SName
